@startuml
actor "Студент" as stu
usecase "Регистрация\n личного кабинета" as reg
usecase "Вход в\n личный кабинет" as log
usecase "Изменение данных в\n личном кабинете" as edit
usecase "Просмотр\n календаря событий" as showEv
usecase "Просмотр\n ленты новостей" as showNews
usecase "Просмотр подробной\n информации о событии" as showMoreEv
usecase "Просмотр подробной\n информации о новости" as showMoreNews
usecase "Добавление события в\nсписок "Желаю посетить"" as addEv
usecase "Управление\n настройками уведомлений" as notify
usecase "Комментирование новости" as comment
usecase "Оценить новость" as like
left to right direction
reg <-- stu
log <-- stu
edit <-- stu
showEv <-- stu
stu --> showNews
stu --> showMoreEv
stu --> showMoreNews
stu --> addEv
notify <- stu
stu -> comment
stu -> like
@enduml
